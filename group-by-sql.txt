1. Contare quanti iscritti ci sono stati ogni anno

SELECT COUNT(*) AS `students_number`, YEAR(`students`.`enrolment_date`) AS `enrollment_year`
FROM `students`
GROUP BY `enrollment_year`;

2. Contare gli insegnanti che hanno l'ufficio nello stesso edificio

SELECT COUNT(*) AS `teachers_number`, `teachers`.`office_address`
FROM `teachers`
GROUP BY `teachers`.`office_address`;

3. Calcolare la media dei voti di ogni appello d'esame

SELECT `exam_student`.`exam_id`, AVG(`exam_student`.`vote`) AS `voto_medio`
FROM `exam_student`
GROUP BY `exam_student`.`exam_id`;

4. Contare quanti corsi di laurea ci sono per ogni dipartimento

SELECT `degrees`.`department_id`, COUNT(*) as `degrees_number` 
FROM `degrees`
GROUP BY `degrees`.`department_id`;
